# Vim

Personal vim configuration.

You can install this script by running the following command in your terminal.

```sh
curl -sL https://MR-Addict.github.io/setup/api/script/vim | bash
```

Below is the raw script content, which you can also get from the command above from the api https://MR-Addict.github.io/setup/api/script/vim.

```sh
#! /bin/bash

# 1. Install Vim
if ! command -v vim &>/dev/null ; then
  echo "[INFO] Installing vim..."
  sudo apt-get install vim -y &>/dev/null
else
  echo "[WARN] You have installed vim!"
fi

# 2. Config vim
if [ ! -f /home/$USER/.vimrc ] ; then
  echo "[INFO] Configurating vim for $USER..."
  wget -q https://MR-Addict.github.io/setup/assets/vim/vimrc -O /home/$USER/.vimrc
else
  echo "[WARN] You have already configurated vim in $USER!"
fi

if ! sudo test -f /root/.vimrc; then
  echo "[INFO] Configurating vim for root..."
  sudo cp /home/$USER/.vimrc /root
else
  echo "[WARN] You have already configurated vim in root!"
fi

echo "[INFO] Configuting vim as your default editor..."
sudo update-alternatives --set editor /usr/bin/vim.basic &>/dev/null
